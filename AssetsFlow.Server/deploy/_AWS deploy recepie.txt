cp /mnt/d/tmp2/HsR_AF.pem ~/.ssh/
chmod 400 ~/.ssh/your-key.pem

2. scp required files
scp -i ~/.ssh/HsR_AF.pem /mnt/d/tmp/* ubuntu@13.61.16.230:/home/ubuntu/

3. establish ssh, check files (-a show hidden, which .env should be)
ssh -i ~/.ssh/HsR_AF.pem ubuntu@13.61.16.230
ls -a /home/ubuntu/


4. fix windows line endings, and execute
sed -i 's/\r//' deploy.sh
./deploy.sh

exit
